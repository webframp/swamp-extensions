# Cloudflare Audit Extension Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build a `@webframp/cloudflare-audit` extension with a workflow that gathers data from 5 existing Cloudflare models and a report that analyzes findings by severity.

**Architecture:** New extension directory `cloudflare-audit/` with a workflow-scope report following the SRE health check pattern. Parallel workflow jobs call existing `@webframp/cloudflare/*` model methods, a finalize job gates completion, and the report reads step outputs via `dataRepository.getContent()` using the string-then-type-object fallback.

**Tech Stack:** TypeScript (Deno), swamp workflow YAML, `@systeminit/swamp-testing` for report tests

---

## Context for Implementers

### Key patterns from the SRE health check to replicate

**Data access in workflow-scope reports:**

The report receives `context.stepExecutions` — an array of step results. Each step has `modelName`, `methodName`, `modelType`, `modelId`, `status`, and `dataHandles[]`. The report uses two helpers:

1. `findStepData(modelName, methodName)` — searches `stepExecutions` for a matching model+method, returns the first non-report data handle's `{ modelType, modelId, dataName, version }`.

2. `getData(modelType, modelId, dataName, version)` — calls `dataRepository.getContent()` with a string first (works in tests), then falls back to a type-like object `{ raw, toDirectoryPath: () => modelType, toString: () => modelType }` (works in real runtime). Parses the `Uint8Array` result as JSON.

**Cloudflare model data shapes (what the report will read):**

| Workflow Step | Model Instance | Method | Spec Written | Instance Name | Key Fields |
|---------------|---------------|--------|-------------|--------------|------------|
| list-zones | cf-zone | list | zones | "main" | zones[].{id, name, status, paused, development_mode} |
| get-settings | cf-zone | get_settings | settings | {zoneId} | settings.{ssl, always_use_https, ...} |
| list-rules | cf-waf | list_rules | rules | "main" | rules[].{id, paused, action, description, filter.expression} |
| list-packages | cf-waf | list_packages | packages | "main" | packages[].{id, name, detection_mode} |
| security-events | cf-waf | get_security_events | events | "recent" | events[].{action, source, clientIP} |
| list-dns | cf-dns | list | records | "main" | records[].{id, name, type, content, proxied, ttl} |
| list-workers | cf-worker | list_scripts | scripts | "main" | scripts[].{id, created_on, modified_on, handlers} |
| list-routes | cf-worker | list_routes | routes | {zoneId} | routes[].{id, pattern, script} |
| cache-settings | cf-cache | get_settings | settings | "main" | cacheLevel, browserCacheTtl, developmentMode, alwaysOnline |
| cache-analytics | cf-cache | get_analytics | analytics | "main" | requests.{all, cached, uncached, cacheHitRate}, bandwidth |
| zone-detail | cf-zone | get | zone | {zoneId} | id, name, status, paused, plan, name_servers |

### Report `findStepData()` matching

The report matches on `modelName` (the instance name, e.g., `"cf-zone"`) and `methodName` (e.g., `"list"`). Each step in `stepExecutions` carries the `modelType` (e.g., `"@webframp/cloudflare/zone"`) which is used for `getContent()`.

```typescript
// Cloudflare Audit Report
// SPDX-License-Identifier: Apache-2.0

interface DataHandle {
  name: string;
  dataId: string;
  version: number;
}

interface StepExecution {
  jobName: string;
  stepName: string;
  modelName: string;
  modelType: string;
  modelId: string;
  methodName: string;
  status: string;
  dataHandles: DataHandle[];
}

interface DataRepository {
  getContent(
    modelType: unknown,
    modelId: string,
    dataName: string,
    version?: number,
  ): Promise<Uint8Array | null>;
}

interface WorkflowReportContext {
  workflowId: string;
  workflowRunId: string;
  workflowName: string;
  workflowStatus: string;
  stepExecutions: StepExecution[];
  repoDir: string;
  dataRepository: DataRepository;
  logger: {
    info: (msg: string, props: Record<string, unknown>) => void;
  };
}

// --- Thresholds ---

export const CACHE_HIT_RATE_WARN = 50;

// Dangling CNAME patterns — CNAMEs pointing at these may be vulnerable
// to subdomain takeover if the backing resource is deprovisioned.
export const DANGLING_CNAME_PATTERNS = [
  /\.herokuapp\.com$/,
  /\.herokudns\.com$/,
  /\.s3\.amazonaws\.com$/,
  /\.s3-website[.-].*\.amazonaws\.com$/,
  /\.cloudfront\.net$/,
  /\.azurewebsites\.net$/,
  /\.blob\.core\.windows\.net$/,
  /\.trafficmanager\.net$/,
  /\.cloudapp\.net$/,
  /\.github\.io$/,
  /\.shopify\.com$/,
  /\.pantheonsite\.io$/,
  /\.ghost\.io$/,
  /\.netlify\.app$/,
  /\.fly\.dev$/,
  /\.vercel\.app$/,
];

type Severity = "ok" | "warn" | "critical" | "error";

interface Finding {
  check: string;
  severity: Severity;
  message: string;
  detail?: string;
}

// --- Data access helpers ---

function findStepData(
  stepExecutions: StepExecution[],
  modelName: string,
  methodName: string,
): {
  modelType: string;
  modelId: string;
  dataName: string;
  version: number;
} | null {
  for (const step of stepExecutions) {
    if (step.modelName === modelName && step.methodName === methodName) {
      for (const handle of step.dataHandles) {
        if (!handle.name.startsWith("report-")) {
          return {
            modelType: step.modelType,
            modelId: step.modelId,
            dataName: handle.name,
            version: handle.version,
          };
        }
      }
    }
  }
  return null;
}

async function getData(
  repo: DataRepository,
  modelType: string,
  modelId: string,
  dataName: string,
  version: number,
): Promise<Record<string, unknown> | null> {
  try {
    const raw = await repo.getContent(modelType, modelId, dataName, version);
    if (raw) return JSON.parse(new TextDecoder().decode(raw));
  } catch {
    try {
      const typeArg = {
        raw: modelType,
        toDirectoryPath: () => modelType,
        toString: () => modelType,
      };
      const raw = await repo.getContent(typeArg, modelId, dataName, version);
      if (raw) return JSON.parse(new TextDecoder().decode(raw));
    } catch {
      // both approaches failed
    }
  }
  return null;
}

// --- Audit check functions (exported for testing) ---

export function checkZone(
  zoneData: Record<string, unknown> | null,
  settingsData: Record<string, unknown> | null,
): Finding[] {
  const findings: Finding[] = [];

  if (!zoneData && !settingsData) {
    findings.push({
      check: "Zone",
      severity: "error",
      message: "No zone data available",
    });
    return findings;
  }

  // Zone paused
  if (zoneData) {
    const zones = zoneData.zones as Array<Record<string, unknown>> | undefined;
    // If we got zone list data, check all zones; otherwise check single zone
    if (zones) {
      for (const z of zones) {
        if (z.paused === true) {
          findings.push({
            check: "Zone Paused",
            severity: "critical",
            message: `Zone "${z.name}" is paused — Cloudflare proxy disabled`,
          });
        }
        if (z.status !== "active") {
          findings.push({
            check: "Zone Status",
            severity: "warn",
            message: `Zone "${z.name}" status is "${z.status}" (not active)`,
          });
        }
      }
    }
  }

  // Zone settings checks
  if (settingsData) {
    const settings = settingsData.settings as Record<string, unknown>;
    if (settings) {
      // SSL mode
      const ssl = settings.ssl as string | undefined;
      if (ssl === "off") {
        findings.push({
          check: "SSL Mode",
          severity: "critical",
          message: "SSL is disabled — all traffic is unencrypted",
        });
      } else if (ssl === "flexible") {
        findings.push({
          check: "SSL Mode",
          severity: "warn",
          message:
            'SSL set to "flexible" — origin traffic is unencrypted',
        });
      } else if (ssl) {
        findings.push({
          check: "SSL Mode",
          severity: "ok",
          message: `SSL mode: ${ssl}`,
        });
      }

      // Always Use HTTPS
      const alwaysHttps = settings.always_use_https as string | undefined;
      if (alwaysHttps === "off") {
        findings.push({
          check: "Always Use HTTPS",
          severity: "warn",
          message: "Always Use HTTPS is disabled",
        });
      } else if (alwaysHttps === "on") {
        findings.push({
          check: "Always Use HTTPS",
          severity: "ok",
          message: "Always Use HTTPS is enabled",
        });
      }

      // Development mode
      const devMode = settings.development_mode as number | string | undefined;
      if (devMode && devMode !== 0 && devMode !== "off") {
        findings.push({
          check: "Development Mode",
          severity: "critical",
          message: "Development mode is enabled — caching bypassed",
        });
      }
    }
  }

  return findings;
}

export function checkWaf(
  rulesData: Record<string, unknown> | null,
  packagesData: Record<string, unknown> | null,
  _eventsData: Record<string, unknown> | null,
): Finding[] {
  const findings: Finding[] = [];

  // WAF rules
  if (rulesData) {
    const rules = rulesData.rules as Array<Record<string, unknown>>;
    const activeRules = rules.filter((r) => r.paused !== true);
    const pausedRules = rules.filter((r) => r.paused === true);

    if (activeRules.length === 0 && pausedRules.length === 0) {
      findings.push({
        check: "WAF Rules",
        severity: "warn",
        message: "No firewall rules configured",
      });
    } else if (activeRules.length === 0) {
      findings.push({
        check: "WAF Rules",
        severity: "warn",
        message: `All ${pausedRules.length} firewall rule(s) are paused`,
      });
    } else {
      findings.push({
        check: "WAF Rules",
        severity: "ok",
        message: `${activeRules.length} active firewall rule(s)`,
      });
    }

    if (pausedRules.length > 0) {
      for (const rule of pausedRules) {
        findings.push({
          check: "Paused Rule",
          severity: "warn",
          message: `Firewall rule "${
            rule.description || rule.id
          }" is paused`,
        });
      }
    }
  } else {
    findings.push({
      check: "WAF Rules",
      severity: "error",
      message: "Could not retrieve firewall rules",
    });
  }

  // WAF packages
  if (packagesData) {
    const packages = packagesData.packages as Array<Record<string, unknown>>;
    if (packages.length === 0) {
      findings.push({
        check: "WAF Packages",
        severity: "warn",
        message: "No managed WAF rulesets found",
      });
    } else {
      findings.push({
        check: "WAF Packages",
        severity: "ok",
        message: `${packages.length} managed WAF ruleset(s) active`,
      });
    }
  }

  return findings;
}

export function checkDns(
  recordsData: Record<string, unknown> | null,
): Finding[] {
  const findings: Finding[] = [];

  if (!recordsData) {
    findings.push({
      check: "DNS",
      severity: "error",
      message: "Could not retrieve DNS records",
    });
    return findings;
  }

  const records = recordsData.records as Array<Record<string, unknown>>;

  // Unproxied A/AAAA/CNAME records
  const proxyableTypes = ["A", "AAAA", "CNAME"];
  const unproxied = records.filter(
    (r) => proxyableTypes.includes(r.type as string) && r.proxied === false,
  );
  if (unproxied.length > 0) {
    const names = unproxied.map((r) => r.name as string);
    const display = names.length <= 5
      ? names.join(", ")
      : `${names.slice(0, 5).join(", ")} (+${names.length - 5} more)`;
    findings.push({
      check: "Unproxied Records",
      severity: "warn",
      message:
        `${unproxied.length} A/AAAA/CNAME record(s) expose origin IP`,
      detail: display,
    });
  } else {
    const proxyable = records.filter((r) =>
      proxyableTypes.includes(r.type as string)
    );
    if (proxyable.length > 0) {
      findings.push({
        check: "Unproxied Records",
        severity: "ok",
        message: `All ${proxyable.length} proxyable record(s) are proxied`,
      });
    }
  }

  // Dangling CNAMEs
  const cnames = records.filter((r) => r.type === "CNAME");
  for (const cname of cnames) {
    const content = cname.content as string;
    const isDangling = DANGLING_CNAME_PATTERNS.some((p) => p.test(content));
    if (isDangling) {
      findings.push({
        check: "Dangling CNAME",
        severity: "critical",
        message:
          `${cname.name} points to ${content} — potential subdomain takeover`,
      });
    }
  }

  // Missing CAA records
  const hasCaa = records.some((r) => r.type === "CAA");
  if (!hasCaa) {
    findings.push({
      check: "CAA Records",
      severity: "warn",
      message: "No CAA records — any CA can issue certificates for this zone",
    });
  } else {
    findings.push({
      check: "CAA Records",
      severity: "ok",
      message: "CAA records present",
    });
  }

  return findings;
}

export function checkWorkers(
  scriptsData: Record<string, unknown> | null,
  routesData: Record<string, unknown> | null,
): Finding[] {
  const findings: Finding[] = [];

  if (!scriptsData) return findings; // Workers are optional

  const scripts = scriptsData.scripts as Array<Record<string, unknown>>;
  if (scripts.length === 0) return findings;

  const routes = routesData
    ? (routesData.routes as Array<Record<string, unknown>>)
    : [];

  // Workers without routes (orphaned)
  const routedScripts = new Set(
    routes.map((r) => r.script as string).filter(Boolean),
  );
  const orphaned = scripts.filter((s) => !routedScripts.has(s.id as string));
  if (orphaned.length > 0) {
    findings.push({
      check: "Orphaned Workers",
      severity: "warn",
      message: `${orphaned.length} Worker script(s) have no routes`,
      detail: orphaned.map((s) => s.id as string).join(", "),
    });
  } else {
    findings.push({
      check: "Workers",
      severity: "ok",
      message:
        `All ${scripts.length} Worker script(s) have routes`,
    });
  }

  return findings;
}

export function checkCache(
  settingsData: Record<string, unknown> | null,
  analyticsData: Record<string, unknown> | null,
): Finding[] {
  const findings: Finding[] = [];

  // Cache level
  if (settingsData) {
    const level = settingsData.cacheLevel as string | undefined;
    if (level === "bypass" || level === "basic") {
      findings.push({
        check: "Cache Level",
        severity: "warn",
        message: `Cache level set to "${level}" — consider "aggressive"`,
      });
    } else if (level) {
      findings.push({
        check: "Cache Level",
        severity: "ok",
        message: `Cache level: ${level}`,
      });
    }
  }

  // Cache analytics
  if (analyticsData) {
    const requests = analyticsData.requests as Record<string, unknown>;
    if (requests) {
      const hitRate = requests.cacheHitRate as number | undefined;
      if (hitRate !== undefined && hitRate < CACHE_HIT_RATE_WARN) {
        findings.push({
          check: "Cache Hit Rate",
          severity: "warn",
          message: `Cache hit rate is ${hitRate.toFixed(1)}% — below ${CACHE_HIT_RATE_WARN}% threshold`,
        });
      } else if (hitRate !== undefined) {
        findings.push({
          check: "Cache Hit Rate",
          severity: "ok",
          message: `Cache hit rate: ${hitRate.toFixed(1)}%`,
        });
      }
    }
  }

  return findings;
}

// --- Report definition ---

export const report = {
  name: "@webframp/cloudflare-audit-report",
  description:
    "Analyzes Cloudflare zone settings, DNS, WAF, Workers, and cache for security and configuration issues",
  scope: "workflow" as const,
  labels: ["cloudflare", "security", "audit"],

  execute: async (context: WorkflowReportContext) => {
    const findings: Finding[] = [];
    const jsonData: Record<string, unknown> = {
      workflowName: context.workflowName,
      workflowStatus: context.workflowStatus,
      timestamp: new Date().toISOString(),
    };

    // --- Load step data ---

    async function loadStep(
      modelName: string,
      methodName: string,
    ): Promise<Record<string, unknown> | null> {
      const loc = findStepData(context.stepExecutions, modelName, methodName);
      if (!loc) return null;
      return await getData(
        context.dataRepository,
        loc.modelType,
        loc.modelId,
        loc.dataName,
        loc.version,
      );
    }

    const zoneListData = await loadStep("cf-zone", "list");
    const settingsData = await loadStep("cf-zone", "get_settings");
    const rulesData = await loadStep("cf-waf", "list_rules");
    const packagesData = await loadStep("cf-waf", "list_packages");
    const eventsData = await loadStep("cf-waf", "get_security_events");
    const dnsData = await loadStep("cf-dns", "list");
    const scriptsData = await loadStep("cf-worker", "list_scripts");
    const routesData = await loadStep("cf-worker", "list_routes");
    const cacheSettingsData = await loadStep("cf-cache", "get_settings");
    const cacheAnalyticsData = await loadStep("cf-cache", "get_analytics");
    const zoneDetailData = await loadStep("cf-zone", "get");

    // Store raw data in JSON output
    jsonData.zoneList = zoneListData;
    jsonData.zoneSettings = settingsData;
    jsonData.zoneDetail = zoneDetailData;
    jsonData.wafRules = rulesData;
    jsonData.wafPackages = packagesData;
    jsonData.securityEvents = eventsData;
    jsonData.dnsRecords = dnsData;
    jsonData.workerScripts = scriptsData;
    jsonData.workerRoutes = routesData;
    jsonData.cacheSettings = cacheSettingsData;
    jsonData.cacheAnalytics = cacheAnalyticsData;

    // --- Run checks ---

    findings.push(...checkZone(zoneListData, settingsData));
    findings.push(...checkWaf(rulesData, packagesData, eventsData));
    findings.push(...checkDns(dnsData));
    findings.push(...checkWorkers(scriptsData, routesData));
    findings.push(...checkCache(cacheSettingsData, cacheAnalyticsData));

    // --- Build markdown ---

    const criticals = findings.filter((f) => f.severity === "critical");
    const warnings = findings.filter((f) => f.severity === "warn");
    const errors = findings.filter((f) => f.severity === "error");
    const oks = findings.filter((f) => f.severity === "ok");

    const overallStatus = criticals.length > 0
      ? "CRITICAL"
      : warnings.length > 0
      ? "WARNING"
      : errors.length > 0
      ? "DEGRADED"
      : "HEALTHY";

    const statusEmoji: Record<string, string> = {
      CRITICAL: "[CRITICAL]",
      WARNING: "[WARNING]",
      DEGRADED: "[DEGRADED]",
      HEALTHY: "[HEALTHY]",
    };

    // Zone name for header
    const zoneName = zoneDetailData
      ? (zoneDetailData.name as string)
      : settingsData
      ? (settingsData.zoneName as string)
      : "unknown";

    const lines: string[] = [];
    lines.push("# Cloudflare Audit Report");
    lines.push("");
    lines.push(`**Status:** ${statusEmoji[overallStatus]} ${overallStatus}`);
    lines.push(`**Zone:** ${zoneName}`);
    lines.push(`**Timestamp:** ${new Date().toISOString()}`);
    lines.push(
      `**Checks:** ${findings.length} total | ${criticals.length} critical | ${warnings.length} warning | ${oks.length} ok`,
    );
    lines.push("");

    if (criticals.length > 0) {
      lines.push("## Critical Issues");
      lines.push("");
      for (const f of criticals) {
        lines.push(`- **${f.check}**: ${f.message}`);
        if (f.detail) lines.push(`  - ${f.detail}`);
      }
      lines.push("");
    }

    if (warnings.length > 0) {
      lines.push("## Warnings");
      lines.push("");
      for (const f of warnings) {
        lines.push(`- **${f.check}**: ${f.message}`);
        if (f.detail) lines.push(`  - ${f.detail}`);
      }
      lines.push("");
    }

    if (errors.length > 0) {
      lines.push("## Data Errors");
      lines.push("");
      for (const f of errors) {
        lines.push(`- **${f.check}**: ${f.message}`);
      }
      lines.push("");
    }

    lines.push("## All Checks");
    lines.push("");
    lines.push("| Check | Status | Detail |");
    lines.push("|-------|--------|--------|");
    for (const f of findings) {
      const badge = {
        ok: "OK",
        warn: "WARN",
        critical: "CRIT",
        error: "ERR",
      }[f.severity];
      lines.push(`| ${f.check} | ${badge} | ${f.message} |`);
    }
    lines.push("");

    // --- Recommendations ---

    const recommendations: string[] = [];

    for (const f of criticals) {
      if (f.check === "SSL Mode") {
        recommendations.push(
          'Set SSL mode to "full" or "strict" to encrypt origin traffic',
        );
      }
      if (f.check === "Development Mode") {
        recommendations.push(
          "Disable development mode to restore caching",
        );
      }
      if (f.check === "Zone Paused") {
        recommendations.push(
          "Unpause the zone to restore Cloudflare proxy protection",
        );
      }
      if (f.check === "Dangling CNAME") {
        recommendations.push(
          `Verify or remove ${f.message.split(" points to ")[0]} — risk of subdomain takeover`,
        );
      }
    }

    for (const f of warnings) {
      if (f.check === "SSL Mode") {
        recommendations.push(
          'Upgrade SSL mode from "flexible" to "full" or "strict"',
        );
      }
      if (f.check === "Always Use HTTPS") {
        recommendations.push("Enable Always Use HTTPS to force HTTPS redirects");
      }
      if (f.check === "WAF Rules") {
        recommendations.push(
          "Configure firewall rules to protect against common attacks",
        );
      }
      if (f.check === "Paused Rule") {
        recommendations.push(
          `Review paused firewall rule: ${f.message}`,
        );
      }
      if (f.check === "WAF Packages") {
        recommendations.push("Enable managed WAF rulesets for baseline protection");
      }
      if (f.check === "Unproxied Records") {
        recommendations.push(
          "Proxy A/AAAA/CNAME records through Cloudflare to hide origin IPs",
        );
      }
      if (f.check === "CAA Records") {
        recommendations.push(
          "Add CAA records to restrict which CAs can issue certificates",
        );
      }
      if (f.check === "Orphaned Workers") {
        recommendations.push(
          "Review orphaned Worker scripts — delete if unused or add routes",
        );
      }
      if (f.check === "Cache Level") {
        recommendations.push(
          'Set cache level to "aggressive" for better performance',
        );
      }
      if (f.check === "Cache Hit Rate") {
        recommendations.push(
          "Investigate low cache hit rate — review Page Rules and Cache-Control headers",
        );
      }
      if (f.check === "Zone Status") {
        recommendations.push(
          "Complete zone activation to enable Cloudflare features",
        );
      }
    }

    if (recommendations.length > 0) {
      lines.push("## Recommendations");
      lines.push("");
      for (let i = 0; i < recommendations.length; i++) {
        lines.push(`${i + 1}. ${recommendations[i]}`);
      }
      lines.push("");
    }

    jsonData.findings = findings;
    jsonData.overallStatus = overallStatus;
    jsonData.recommendations = recommendations;
    jsonData.zone = zoneName;

    context.logger.info(
      "Cloudflare audit: {status}, {critical} critical, {warn} warnings",
      {
        status: overallStatus,
        critical: criticals.length,
        warn: warnings.length,
      },
    );

    return {
      markdown: lines.join("\n"),
      json: jsonData,
    };
  },
};
```
